function renderOrders() { const orders = getUserOrders(); const container = document.getElementById('ordersList'); if (!container) return; if (orders.length === 0) { container.innerHTML = '
📭 У вас пока нет заказов
'; return; } container.innerHTML = orders.map(order => `
📋 ${order.order_id} 📅 ${order.date || '—'} ${getStatusText(order.status)} ${(order.total || 0).toLocaleString()} ₽
${(order.items || []).map(i => `${escapeHtml(i.title)} x${i.quantity}`).join(', ')}
${order.tracking_number ? `
📦 Трек-номер: ${escapeHtml(order.tracking_number)} 🔍 Отследить
` : ''}
`).join(''); } // Функция для показа деталей заказа (с трек-номером) window.showOrderDetails = function(orderId) { const orders = getUserOrders(); const order = orders.find(o => o.order_id === orderId); if (!order) return; const itemsHtml = (order.items || []).map(item => `
${escapeHtml(item.title)} (${escapeHtml(item.size || '—')}, ${escapeHtml(item.color || '—')}) x${item.quantity} ${(item.price * item.quantity).toLocaleString()} ₽
`).join(''); const modalHtml = `

📋 Заказ ${order.order_id}

📅 Дата: ${order.date || '—'}
📊 Статус: ${getStatusText(order.status)}
🚚 Доставка: ${order.delivery?.method === 'courier' ? 'Курьером' : order.delivery?.method === 'pickup' ? 'Самовывоз' : '—'} (${(order.delivery?.price || 0).toLocaleString()} ₽)
📍 Адрес: ${escapeHtml(order.address || '—')}
${order.tracking_number ? `
📦 Трек-номер: ${escapeHtml(order.tracking_number)} 🔍 Отследить
` : ''}

🛍️ Товары в заказе:

${itemsHtml}
Итого: ${(order.total || 0).toLocaleString()} ₽

💳 Оплата:

${order.payment === 'card' ? '💳 Банковской картой' : order.payment === 'sbp' ? '📱 СБП' : '💰 Наличными при получении'} ${order.payment_status === 'paid' ? ' (Оплачен)' : ' (Ожидает оплаты)'}
${order.comment ? `

💬 Комментарий:

${escapeHtml(order.comment)}
` : ''}
`; // Удаляем старую модалку, если есть const oldModal = document.getElementById('orderDetailsModal'); if (oldModal) oldModal.remove(); document.body.insertAdjacentHTML('beforeend', modalHtml); document.body.style.overflow = 'hidden'; }; window.closeOrderDetails = function() { const modal = document.getElementById('orderDetailsModal'); if (modal) modal.remove(); document.body.style.overflow = ''; };
