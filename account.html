window.showOrderDetails = function(orderId) { const orders = getUserOrders(); const order = orders.find(o => o.order_id === orderId); if (!order) return; // Формируем таблицу товаров с полной информацией const itemsHtml = (order.items || []).map(item => `
${escapeHtml(item.title)}
Арт: ${escapeHtml(item.article || '—')}
${escapeHtml(item.size || '—')}
Размер
${escapeHtml(item.color || '—')}
Цвет
${item.quantity}
Кол-во
${(item.price * item.quantity).toLocaleString()} ₽
${item.price.toLocaleString()} ₽ / шт
`).join(''); // Информация о доставке const deliveryInfo = order.delivery?.method === 'courier' ? '🚚 Курьерская доставка' : order.delivery?.method === 'pickup' ? '🏪 Самовывоз' : '—'; const deliveryPrice = order.delivery?.price || 0; const deliveryAddress = order.address || '—'; // Статус оплаты const paymentStatusHtml = order.payment_status === 'paid' ? '✅ Оплачен' : '⏳ Ожидает оплаты'; // Способ оплаты const paymentMethodHtml = order.payment === 'card' ? '💳 Банковской картой' : order.payment === 'sbp' ? '📱 СБП' : '💰 Наличными при получении'; const modalHtml = `

📋 Заказ ${order.order_id}

📅 Дата:
${order.date || '—'}
📊 Статус:
${getStatusText(order.status)}
🚚 Доставка:
${deliveryInfo} (${deliveryPrice.toLocaleString()} ₽)
📍 Адрес:
${escapeHtml(deliveryAddress)}
${order.tracking_number ? `
📦 Трек-номер для отслеживания:
${escapeHtml(order.tracking_number)}
🔍 Отследить на карте
` : ''}

🛍️ Товары в заказе (${(order.items || []).length} позиций)

Товар
Размер
Цвет
Кол-во
Сумма
${itemsHtml}
Итого к оплате:
${(order.total || 0).toLocaleString()} ₽
💳 Способ оплаты
${paymentMethodHtml}
${paymentStatusHtml}
💰 Детали заказа
Товары: ${(order.subtotal || order.total - deliveryPrice).toLocaleString()} ₽
Доставка: ${deliveryPrice === 0 ? 'Бесплатно' : deliveryPrice.toLocaleString() + ' ₽'}
${order.comment ? `
💬 Комментарий к заказу
${escapeHtml(order.comment)}
` : ''}
`; // Удаляем старую модалку, если есть const oldModal = document.getElementById('orderDetailsModal'); if (oldModal) oldModal.remove(); document.body.insertAdjacentHTML('beforeend', modalHtml); document.body.style.overflow = 'hidden'; };
